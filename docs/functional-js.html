function pickAnotherPerson(evt) {
	evt.preventDefault();
	evt.stopPropagation();
	evt.stopImmediatePropagation();

	var ID = $(evt.target).attr("rel").replace(/^.*(\d+)$/,"$1");
	EVT.emit("person-selected",ID);
}
