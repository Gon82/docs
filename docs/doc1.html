//fn throttle
let throttle = (func, limit) => {
	let inThrottle
	return function () {
		let args = arguments
		let context = this
		if (!inThrottle) {
			func.apply(context, args)
			inThrottle = true
			setTimeout(() => inThrottle = false, limit)
		}
	}
}
